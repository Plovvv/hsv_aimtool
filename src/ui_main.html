  64class GlowLabel(tk.Canvas):
  65    """Компонент текстовой метки с многослойным эффектом неонового свечения."""
  66
  67    def __init__(self, master, text: str, font=FONT_TITLE,
  68                 text_color=C_ACCENT, glow_color=C_ACCENT, **kw):
  69        super().__init__(master, bg=C_BG, highlightthickness=0, **kw)
  70        self._text = text
  71        self._font = font
  72        self._text_color = text_color
  73        self._glow_color = glow_color
  74        self.bind("<Configure>", lambda e: self._draw())
  75        self.after(50, self._draw)
  76
  77    def _draw(self) -> None:
  78        """Выполняет послойную отрисовку размытия тени и основного текста."""
  79        self.delete("all")
  80        w = self.winfo_width() or 300
  81        h = self.winfo_height() or 60
  82
  83        # Генерация эффекта свечения через полупрозрачные смещенные слои
  84        offsets = [3, 2, 1]
  85        alphas = ["20", "40", "60"]
  86        for off, alpha in zip(offsets, alphas):
  87            color = self._glow_color + alpha
  88            try:
  89                self.create_text(w // 2 + off, h // 2 + off,
  90                                 text=self._text, font=self._font,
  91                                 fill=color, anchor="center")
  92                self.create_text(w // 2 - off, h // 2 - off,
  93                                 text=self._text, font=self._font,
  94                                 fill=color, anchor="center")
  95            except Exception:
  96                pass
  97
  98        self.create_text(w // 2, h // 2, text=self._text,
  99                         font=self._font, fill=self._text_color, anchor="center")

 131    def __init__(self, master, label: str, min_val: int, max_val: int,

 137        self._on_change = on_change
 138        self._var_lo = tk.IntVar(value=min_val)

 140        self._from = from_
 141        self._to = to

 143        # Индикатор наименования канала

 145                     text_color=accent, width=28, anchor="w").grid(
 146            row=0, column=0, rowspan=2, padx=(0, 10))

 148        # Слайдер минимального значения (MIN)
 149        ctk.CTkLabel(self, text="MIN", font=FONT_SMALL,
 150                     text_color=C_TEXT_DIM).grid(row=0, column=1, sticky="w")
 151        self._sl_lo = ctk.CTkSlider(
 152            self, from_=from_, to=to, variable=self._var_lo,
 153            progress_color=accent, button_color=accent,
 154            button_hover_color=C_TEXT, fg_color=C_SURFACE2,
 155            command=self._on_slide)
 156        self._sl_lo.grid(row=0, column=2, padx=8, sticky="ew")
 157        self._lbl_lo = ctk.CTkLabel(self, text=str(min_val),
 158                                    font=FONT_LABEL, text_color=C_TEXT, width=38)
 159        self._lbl_lo.grid(row=0, column=3)
 160
 161        # Слайдер максимального значения (MAX)
 162        ctk.CTkLabel(self, text="MAX", font=FONT_SMALL,
 163                     text_color=C_TEXT_DIM).grid(row=1, column=1, sticky="w")
 164        self._sl_hi = ctk.CTkSlider(
 165            self, from_=from_, to=to, variable=self._var_hi,
 166            progress_color=accent, button_color=accent,
 167            button_hover_color=C_TEXT, fg_color=C_SURFACE2,
 168            command=self._on_slide)
 169        self._sl_hi.grid(row=1, column=2, padx=8, sticky="ew")
 170        self._lbl_hi = ctk.CTkLabel(self, text=str(max_val),
 171                                    font=FONT_LABEL, text_color=C_TEXT, width=38)
 172        self._lbl_hi.grid(row=1, column=3)
 173
 174        self.columnconfigure(2, weight=1)
 175
 176    def _on_slide(self, _=None) -> None:
 177        """Синхронизирует текстовые значения с текущим положением ползунков."""
 178        lo = int(self._var_lo.get())
 179        hi = int(self._var_hi.get())
 180        self._lbl_lo.configure(text=str(lo))
 181        self._lbl_hi.configure(text=str(hi))
 182        if self._on_change:
 183            self._on_change()
 184
 185    def get(self) -> tuple:
 186        """Возвращает текущие выбранные границы диапазона (min, max)."""
 187        return int(self._var_lo.get()), int(self._var_hi.get())
 188
 189    def set(self, lo: int, hi: int) -> None:
 190        """Задает границы диапазона программным путем."""
 191        self._var_lo.set(lo)
 192        self._var_hi.set(hi)
 193        self._lbl_lo.configure(text=str(lo))
 194        self._lbl_hi.configure(text=str(hi))

1078    def _draw_annotations(self, result: DetectionResult) -> np.ndarray:
1079        """
1080        Рисует аннотации на оверлей-кадре.
1081
1082        Args:
1083            result: Результат детекции.
1084
1085        Returns:
1086            Кадр BGR с нарисованными рамками, точками и т.д.
1087        """
1088        frame = result.frame_overlay.copy()
1089        fov = config.fov_size
1090
1091        # Линии перекрестия по центру
1092        cx, cy = fov // 2, fov // 2
1093        cv2.line(frame, (cx - 20, cy), (cx + 20, cy), (0, 255, 178), 1)
1094        cv2.line(frame, (cx, cy - 20), (cx, cy + 20), (0, 255, 178), 1)
1095
1096        # FOV-рамка
1097        cv2.rectangle(frame, (2, 2), (fov - 2, fov - 2), (20, 60, 40), 1)
1098
1099        for i, obj in enumerate(result.objects):
1100            is_best = i == 0
1101            color = (0, 255, 140) if is_best else (0, 180, 100)
1102            thickness = 2 if is_best else 1
1103
1104            if config.show_bounding_box:
1105                cv2.rectangle(frame,
1106                              (obj.x, obj.y),
1107                              (obj.x + obj.w, obj.y + obj.h),
1108                              color, thickness)
1109                # Уголки bounding box (красивее чем просто рамка)
1110                lc = 8
1111                for px, py in [(obj.x, obj.y),
1112                                (obj.x + obj.w, obj.y),
1113                                (obj.x, obj.y + obj.h),
1114                                (obj.x + obj.w, obj.y + obj.h)]:
1115                    dx = 1 if px == obj.x else -1
1116                    dy = 1 if py == obj.y else -1
1117                    cv2.line(frame, (px, py), (px + dx * lc, py), color, 2)
1118                    cv2.line(frame, (px, py), (px, py + dy * lc), color, 2)
1119
1120            if config.show_center_dot:
1121                cv2.circle(frame, (obj.cx, obj.cy), 3, color, -1)
1122                if is_best:
1123                    cv2.circle(frame, (obj.cx, obj.cy), 8, color, 1)
1124
1125            # Метка с площадью
1126            if is_best:
1127                cv2.putText(frame, f"#{i+1} {int(obj.area)}px",
1128                            (obj.x, obj.y - 5),
1129                            cv2.FONT_HERSHEY_SIMPLEX, 0.4, color, 1)
1130
1131        return frame
